        function updateTriangle(sliderVal) {
            const totalWidth = 400; 
            const startX = 50;
            const endX = 450;
            const yBase = 250;

            const sliderRatio = sliderVal / 100;
            const splitX = startX + (totalWidth * sliderRatio); 

            const m = splitX - startX;
            const n = endX - splitX;
            const h = Math.sqrt(m * n);
            const yTop = yBase - h;

            polyMain.setAttribute('points', `${startX},${yBase} ${endX},${yBase} ${splitX},${yTop}`);
            
            lineH.setAttribute('x1', splitX);
            lineH.setAttribute('y1', yTop);
            lineH.setAttribute('x2', splitX);
            lineH.setAttribute('y2', yBase);

            const scaleA = 15;
            const scaleM = (m / totalWidth) * scaleA;
            const scaleN = (n / totalWidth) * scaleA;
            const scaleH = Math.sqrt(scaleM * scaleN);
            const scaleB = Math.sqrt(scaleM * scaleA);
            const scaleC = Math.sqrt(scaleN * scaleA);

            labelB.textContent = `b = ${scaleB.toFixed(1)}`;
            labelC.textContent = `c = ${scaleC.toFixed(1)}`;
            labelA.textContent = `a = ${scaleA.toFixed(0)}`;
            labelH.textContent = `h = ${scaleH.toFixed(1)}`;
            labelM.textContent = `m = ${scaleM.toFixed(1)}`;
            labelN.textContent = `n = ${scaleN.toFixed(1)}`;

            labelB.setAttribute('x', `${startX + m/2 - 10}`);
            labelB.setAttribute('y', `${yTop + h/2 + 10}`);
            labelC.setAttribute('x', `${splitX + n/2 + 15}`);
            labelC.setAttribute('y', `${yTop + h/2 + 10}`);
            labelH.setAttribute('x', `${splitX + 10}`);
            labelH.setAttribute('y', `${yTop + h/2}`);
            labelM.setAttribute('x', `${startX + m/2}`);
            labelN.setAttribute('x', `${splitX + n/2}`);

            pythCheck.textContent = `${scaleB.toFixed(1)}² + ${scaleC.toFixed(1)}² = ${scaleA}²`;
            
            calcTezao.innerHTML = `${scaleA} × ${scaleH.toFixed(1)} = ${scaleB.toFixed(1)} × ${scaleC.toFixed(1)} <br class="sm:hidden" /> <span class="text-indigo-600 font-semibold">(${(scaleA * scaleH).toFixed(1)} = ${(scaleB * scaleC).toFixed(1)})</span>`;
            calcTezinho.innerHTML = `${scaleH.toFixed(1)}² = ${scaleM.toFixed(1)} × ${scaleN.toFixed(1)} <br class="sm:hidden" /> <span class="text-indigo-600 font-semibold">(${(scaleH * scaleH).toFixed(1)} = ${(scaleM * scaleN).toFixed(1)})</span>`;
            calcCurvaB.innerHTML = `${scaleB.toFixed(1)}² = ${scaleA} × ${scaleM.toFixed(1)} <br class="sm:hidden" /> <span class="text-indigo-600 font-semibold">(${(scaleB * scaleB).toFixed(1)} = ${(scaleA * scaleM).toFixed(1)})</span>`;
            calcCurvaC.innerHTML = `${scaleC.toFixed(1)}² = ${scaleA} × ${scaleN.toFixed(1)} <br class="sm:hidden" /> <span class="text-indigo-600 font-semibold">(${(scaleC * scaleC).toFixed(1)} = ${(scaleA * scaleN).toFixed(1)})</span>`;

            document.getElementById('check-interact').checked = true;
            document.getElementById('slider-percentage').textContent = `Proporção m/n: ${Math.round(sliderVal)}%`;
        }

        // Função para ajustar sandbox com base nos dados da questão atual
        function applyQuestionDataToSandbox() {
            const q = questionBank[currentQuestionIndex];
            if (q.triangleData && q.triangleData.b && q.triangleData.c) {
                const b = q.triangleData.b;
                const c = q.triangleData.c;
                // Evitar divisão por zero
                if (c === 0) return;
                const R = (b / c) ** 2;
                const sliderRatio = R / (1 + R);
                let sliderVal = Math.round(sliderRatio * 100);
                // Limitar ao intervalo do slider
                sliderVal = Math.max(15, Math.min(85, sliderVal));
                slider.value = sliderVal;
                updateTriangle(sliderVal);
            }
        }

        // Função acionada pelo botão "Revisar no Laboratório"
        function reviewInLab() {
            switchTab('sandbox');
            applyQuestionDataToSandbox();
        }

        }

        function selectOption(idx) {
            const q = questionBank[currentQuestionIndex];
            const feedbackBox = document.getElementById('quiz-feedback');
            const feedbackIconContainer = document.getElementById('feedback-icon-container');
            const feedbackIcon = document.getElementById('feedback-icon');
            const feedbackTitle = document.getElementById('feedback-title');
            const feedbackText = document.getElementById('feedback-text');
            const feedbackExplanation = document.getElementById('feedback-explanation');

            feedbackBox.classList.remove('hidden');
            
            if (idx === q.correct) {
                feedbackBox.className = "mt-6 p-5 rounded-2xl border border-emerald-200 bg-emerald-50/70 flex gap-3 items-start transition-all duration-300";
                feedbackIconContainer.className = "p-2.5 rounded-xl bg-emerald-600 text-white shadow-md shadow-emerald-200";
                feedbackIcon.setAttribute('data-lucide', 'check');
                feedbackTitle.textContent = "Incrível! Resposta Exata!";
                feedbackText.textContent = "Você compreendeu a geometria ativa por trás do problema perfeitamente.";
                feedbackExplanation.innerHTML = `<strong>Demonstração passo a passo:</strong> <code class="bg-emerald-100/80 px-1.5 py-0.5 rounded font-mono text-emerald-950">${q.formula}</code><br><br>${q.explanation}`;
                
                xp += 25;
                updateXPDisplay();
                
                document.getElementById('check-answer').checked = true;
            } else {
                feedbackBox.className = "mt-6 p-5 rounded-2xl border border-rose-200 bg-rose-50/70 flex gap-3 items-start transition-all duration-300";
                feedbackIconContainer.className = "p-2.5 rounded-xl bg-rose-600 text-white shadow-md shadow-rose-200";
                feedbackIcon.setAttribute('data-lucide', 'alert-circle');
                feedbackTitle.textContent = "Houve um detalhe na conta!";
                feedbackText.textContent = "Mas errar é parte crucial do aprendizado ativo! Veja a dica abaixo.";
                feedbackExplanation.innerHTML = `<strong>Dica do Professor:</strong> ${q.hint}.<br>Pressione avançar para reiniciar ou use o botão "Revisar no Laboratório" para visualizar as proporções.`;
            }
            lucide.createIcons();
        }

        function nextQuestion() {
            currentQuestionIndex = (currentQuestionIndex + 1) % totalQuestions;
            loadQuestion();
        }

        function updateXPDisplay() {
            document.getElementById('header-xp').textContent = xp;
            if (xp >= 75) {
                currentLevel = 4;
                document.getElementById('header-level').textContent = "Nível 4 (Mestre Curió)";
            } else if (xp >= 50) {
                currentLevel = 3;
                document.getElementById('header-level').textContent = "Nível 3 (Intermediário)";
            } else if (xp >= 25) {
                currentLevel = 2;
                document.getElementById('header-level').textContent = "Nível 2 (Iniciado)";
            }
        }

        function setConfidence(level) {
            selectedConfidence = level;
            const status = document.getElementById('confidence-status');
            const buttons = document.querySelectorAll('.confidence-btn');
            
            buttons.forEach(btn => {
                btn.classList.remove('ring-2', 'ring-offset-2', 'ring-indigo-600');
            });

            if (level === 'Baixo') {
                status.innerHTML = `Confiança: <span class="font-bold text-rose-600">Frágil</span>. Sugerimos explorar mais o Laboratório Dinâmico.`;
            } else if (level === 'Médio') {
                status.innerHTML = `Confiança: <span class="font-bold text-amber-600">OK</span>. Você está no caminho certo para o domínio!`;
            } else {
                status.innerHTML = `Confiança: <span class="font-bold text-emerald-600">Dominado</span>. Excelente dedicação ativa!`;
            }
        }
